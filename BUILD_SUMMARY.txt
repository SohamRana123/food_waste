QUICK COMMANDS
─────────────────────────────────────────────────────────────────────────────

# Start the system
docker-compose up

# Open in browser
http://localhost:3000

# View API health
curl http://localhost:8000/health

# List available kitchens
curl http://localhost:8000/kitchens

# Make a prediction
curl -X POST http://localhost:8000/predict \
  -H "Content-Type: application/json" \
  -d '{"kitchen_id":"K1","forecast_start_date":"2024-04-05","horizon_days":1,...}'

# Get metrics
curl http://localhost:8000/metrics

# Stop everything
docker-compose down
